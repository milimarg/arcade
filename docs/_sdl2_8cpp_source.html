376  void Sdl2::pollEvents()
377  {
378  SDL_Event e;
379 
380  while (SDL_PollEvent(&e) != 0) {
381  if (e.type == SDL_QUIT) {
382  _window->closeWindow();
383  continue;
384  }
385  if (e.type == SDL_KEYDOWN) {
386  SDL_KeyCode currentKey = static_cast<SDL_KeyCode>(e.key.keysym.sym);
387 
388  if (_keyTable.contains(currentKey))
389  _event->pushEvent(_keyTable.at(currentKey));
390 
391  if (_changeKeyTable.contains(currentKey))
392  _event->pushChangeEvent(_changeKeyTable.at(currentKey));
393  continue;
394  }
395  if (e.type == SDL_MOUSEBUTTONDOWN) {
396  if (e.button.button == SDL_BUTTON_LEFT) {
397  _event->pushEvent(Key::MOUSE_BUTTON_LEFT, {e.button.x / std::max(1, static_cast<Window*>(_window.get())->getCurrentPixelSizeX()), e.button.y / std::max(1, static_cast<Window*>(_window.get())->getCurrentPixelSizeY())});
398  } else if (e.button.button == SDL_BUTTON_RIGHT) {
399  _event->pushEvent(Key::MOUSE_BUTTON_RIGHT, {e.button.x / std::max(1, static_cast<Window*>(_window.get())->getCurrentPixelSizeX()), e.button.y / std::max(1, static_cast<Window*>(_window.get())->getCurrentPixelSizeY())});
400  }
401  }
402  }
403  }
