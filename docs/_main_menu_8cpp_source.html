496  std::string MainMenu::decryptData(const std::string &ciphertext, const std::string &key)
497  {
498  OpenSSL_add_all_algorithms();
499  ERR_load_crypto_strings();
500 
501  const EVP_CIPHER *cipher = EVP_aes_256_cbc();
502  const int keyLength = EVP_CIPHER_key_length(cipher);
503  const int ivLength = EVP_CIPHER_iv_length(cipher);
504 
505  std::string ivStr = ciphertext.substr(0, ivLength);
506  std::vector<unsigned char> iv(ivStr.begin(), ivStr.end());
507  EVP_CIPHER_CTX *ctx = EVP_CIPHER_CTX_new();
508  EVP_DecryptInit_ex(ctx, cipher, nullptr, reinterpret_cast<const unsigned char *>(key.c_str()), iv.data());
509  int plainTextLen = ciphertext.length() - ivLength + EVP_CIPHER_block_size(cipher);
510  std::vector<unsigned char> plainText(plainTextLen);
511  int len = 0;
512  EVP_DecryptUpdate(ctx, plainText.data(), &len, reinterpret_cast<const unsigned char *>(ciphertext.c_str() + ivLength), ciphertext.length() - ivLength);
513  plainTextLen = len;
514  EVP_DecryptFinal_ex(ctx, plainText.data() + len, &len);
515  plainTextLen += len;
516  EVP_CIPHER_CTX_free(ctx);
517  return std::string(reinterpret_cast<char*>(plainText.data()), plainTextLen);
518  }
