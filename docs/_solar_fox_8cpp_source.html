286  _pixels->getPixelMap().clear();
287 
288  if (_timeBank - latest > 0.03125f) { // TODO: 0.125 / 4
289  latest = _timeBank;
290  for (auto it = _enemies.begin(); it < _enemies.end(); it += 1)
291  detectWallCollision(*it, (it - _enemies.begin()) % 2 == 0);
292  for (Bullet &bullet : _bullets) {
293  if (!bullet.isActive())
294  continue;
295  if (bullet.getOrientation() == Entity::Orientation::HORIZONTAL)
296  bullet.setX(bullet.getX() + bullet.getShift());
297  else
298  bullet.setY(bullet.getY() + bullet.getShift());
299  detectBulletOutside(bullet);

301  continue;
302  if (bullet.getX() == _player.getX() && bullet.getY() == _player.getY()) {

304  }

306  bullet.getY() == _playerBullet.getY())

308  }
309  }

313  size_t index = it - _bullets.begin();

315  }
316  }
317  spawnPlayerBullet();
318  if (_timeBank - latestPlayer > (_moveFaster ? 0.0625f : 0.125f)) { // TODO: 0.125 / 2 or 0.125

328  detectPlayerBulletTooFar();
329
