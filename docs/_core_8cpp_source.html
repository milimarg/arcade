239  int Core::handleChangeEvent()

245 
246  while (1) {

248  std::unique_ptr<IEventInfo> tempEventInfo = _eventWrapper->getChangeEvent();
249  if (!tempEventInfo)
250  break;
251  tempChangeKey = tempEventInfo->getChangeKey();
252  } catch (const std::exception& e) {
253  throw arcade::Error(e.what());
254  }
255  if (tempChangeKey == ChangeKey::CHANGE_LIST_END || tempChangeKey == ChangeKey::NONE)
256  break;
257  keyList.push_back(tempChangeKey);
258  }
259 
260  for (auto &changeKey : keyList) {
261  switch (changeKey) {
262  case ChangeKey::PREV_GRAPHICAL_LIB:
263  if (_graphicalLibraryList.size() == 0)
264  break;
265  _graphicalIndex = (_graphicalIndex > 0) ? _graphicalIndex - 1 : _graphicalLibraryList.size() - 1;
266  index = _graphicalIndex % _graphicalLibraryList.size();
267  switchGraphicalLibrary(_graphicalLibraryList[index], true);
268  returnValue = 1;
269  break;
270  case ChangeKey::NEXT_GRAPHICAL_LIB:
271  if (_graphicalLibraryList.size() == 0)
272  break;
273  _graphicalIndex = (_graphicalIndex < _graphicalLibraryList.size() - 1) ? _graphicalIndex + 1 : 0;
274  index = _graphicalIndex % _graphicalLibraryList.size();
275  switchGraphicalLibrary(_graphicalLibraryList[index], true);
276  returnValue = 1;
277  break;
278  case ChangeKey::PREV_GAME_LIB:
279  if (_gameLibraryList.size() == 0)
280  break;
281  if (_gameIndex % _gameLibraryList.size() != _indexMainMenu)
282  saveCurrentScore(_gameLibraryList[_gameIndex % _gameLibraryList.size()]);
283  if (_gameIndex < 0)
284  _gameIndex = _gameLibraryList.size() - 1;
285  else
286  _gameIndex = (_gameIndex > 0) ? _gameIndex - 1 : _gameLibraryList.size() - 1;
287  index = _gameIndex % _gameLibraryList.size();

289  returnValue = 1;
290  break;
291  case ChangeKey::NEXT_GAME_LIB:
292  if (_gameLibraryList.size() == 0)
293  break;
294  if (_gameIndex % _gameLibraryList.size() != _indexMainMenu)
295  saveCurrentScore(_gameLibraryList[_gameIndex % _gameLibraryList.size()]);
296  if (_gameIndex < 0)
297  _gameIndex = 0;
298  else
299  _gameIndex = (_gameIndex < _gameLibraryList.size() - 1) ? _gameIndex + 1 : 0;
300  index = _gameIndex % _gameLibraryList.size();

302  returnValue = 1;

305  if (_gameIndex % _gameLibraryList.size() != _indexMainMenu)
306  saveCurrentScore(_gameLibraryList[_gameIndex % _gameLibraryList.size()]);
307  switchMainMenu(true);
308  returnValue = 1;

310  case ChangeKey::ESCAPE:
311  if (_gameIndex % _gameLibraryList.size() != _indexMainMenu)
312  saveCurrentScore(_gameLibraryList[_gameIndex % _gameLibraryList.size()]);
313  _windowWrapper->closeWindow();

315  default:

317  }

445  std::string Core::encryptData(const std::string &plaintext, const std::string &key)
446  {
447  OpenSSL_add_all_algorithms();
448  ERR_load_crypto_strings();
449 
450  const EVP_CIPHER *cipher = EVP_aes_256_cbc();
451  const int keyLength = EVP_CIPHER_key_length(cipher);
452  const int ivLength = EVP_CIPHER_iv_length(cipher);
453 
454  std::vector<unsigned char> iv(ivLength);
455  RAND_bytes(iv.data(), ivLength);
456  EVP_CIPHER_CTX *ctx = EVP_CIPHER_CTX_new();
457  EVP_EncryptInit_ex(ctx, cipher, nullptr, reinterpret_cast<const unsigned char *>(key.c_str()), iv.data());
458  int cipherTextLen = plaintext.length() + EVP_CIPHER_block_size(cipher);
459  std::vector<unsigned char> cipherText(cipherTextLen);
460  int len = 0;
461  EVP_EncryptUpdate(ctx, cipherText.data(), &len, reinterpret_cast<const unsigned char *>(plaintext.c_str()), plaintext.length());
462  cipherTextLen = len;
463  EVP_EncryptFinal_ex(ctx, cipherText.data() + len, &len);
464  cipherTextLen += len;
465  EVP_CIPHER_CTX_free(ctx);
466  std::string result(reinterpret_cast<char*>(iv.data()), iv.size());
467  result += std::string(reinterpret_cast<char*>(cipherText.data()), cipherTextLen);
