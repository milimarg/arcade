460  void Snake::handleCollisions()
461  {
462  std::pair<int, int> head = _snake.getHead();
463  int headX = head.first;
464  int headY = head.second;
465 
466  int arenaWidth = _gameSettings["arenaWidth"];
467  int arenaHeight = _gameSettings["arenaHeight"];
468 
469  if (headX <= 0 || headX >= arenaWidth - 1 || headY <= 0 || headY >= arenaHeight - 1) {
470  _isGameOver = true;
471  _beforeReload = 3.0f;
472  return;
473  }
474 
475  for (auto &food : _foodMap) {
476  if (food.getPosition() == std::make_pair(headX, headY)) {
477  _score += food.getScore();
478 
479  food.applyEffect(_snake);
480 
481  const auto &foodPosition = std::find(_foodMap.begin(), _foodMap.end(), food);
482  _foodMap.erase(foodPosition);
483 
484  spawnFood();
485 
486  return;
487  }
488  }
489 
490  if (_snake.collideWithBody(head)) {
491  _isGameOver = true;
492  _beforeReload = 3.0f;
493  return;
494  }
495  }
