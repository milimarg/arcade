464  void Nibbler::handleCollisions()
465  {
466  std::pair<int, int> head = _snake.getHead();
467  int headX = head.first;

469 
470  int arenaWidth = _gameSettings["arenaWidth"];

472 
473  for (auto &food : _foodMap) {
474  if (food.getPosition() == std::make_pair(headX, headY)) {
475  _score += food.getScore();
476 
477  food.applyEffect(_snake);
478 
479  const auto &foodPosition = std::find(_foodMap.begin(), _foodMap.end(), food);
480  _foodMap.erase(foodPosition);
481 
482  if (!_foodMap.size()) {
483  _gameSettings["foodQuantity"] += 1 + _gameSettings["foodQuantity"] / 10;
484  reload();
485  _snake.increaseBaseMoveSpeed(0.02f);
486  }
487  return;
488  }

490

492  _isGameOver = true;
