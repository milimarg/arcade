146  double Window::getDistanceColor(int r, int g, int b, Pixel pixel)
147  {
148  int red = pixel.getRed();
149  int green = pixel.getGreen();
150  int blue = pixel.getBlue();
151 
152  int dr = r - red;
153  int dg = g - green;
154  int db = b - blue;
155  double distance = std::sqrt(dr * dr + dg * dg + db * db);
156 
157  return distance;

161  {
